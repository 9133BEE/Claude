//@version=5 indicator("盤面趨勢總覽", overlay=true) // ── 參數 ── ema_fast = input.int(9, "快線 EMA") ema_slow = input.int(21, "慢線 EMA") rsi_len = input.int(14, "RSI 週期") adx_len = input.int(14, "ADX 週期") // ── 計算 ── ema9 = ta.ema(close, ema_fast) ema21 = ta.ema(close, ema_slow) vwap = ta.vwap(close) rsi = ta.rsi(close, rsi_len) [di_p, di_m, adx] = ta.dmi(adx_len, adx_len) obv = ta.obv atr = ta.atr(14) // ── 多空判斷 ── trend_ema = ema9 > ema21 ? "多▲" : "空▼" trend_vwap = close > vwap ? "多▲" : "空▼" rsi_str = rsi >= 50 ? "多▲" : "空▼" adx_str = adx >= 25 ? (di_p > di_m ? "強多" : "強空") : "震盪" obv_str = obv > obv[1] ? "買壓" : "賣壓" atr_str = str.tostring(math.floor(atr * 100) / 100) col_bull = color.new(color.green, 0) col_bear = color.new(color.red, 0) col_neu = color.new(color.gray, 0) f_col(s) => s == "多▲" or s == "強多" or s == "買壓" ? col_bull : s == "空▼" or s == "強空" or s == "賣壓" ? col_bear : col_neu // ── Table ── var t = table.new(position.top_right, 2, 7, bgcolor=#161b22, border_color=#30363d, border_width=1, frame_color=#30363d, frame_width=1) f_cell(r, lbl, val, vc) => table.cell(t, 0, r, lbl, text_color=#8b949e, text_size=size.small, bgcolor=#1a1f2e) table.cell(t, 1, r, val, text_color=vc, text_size=size.small, bgcolor=#161b22) f_cell(0, "EMA 趨勢", trend_ema, f_col(trend_ema)) f_cell(1, "VWAP", trend_vwap, f_col(trend_vwap)) f_cell(2, "RSI", rsi_str, f_col(rsi_str)) f_cell(3, "ADX", adx_str, f_col(adx_str)) f_cell(4, "OBV", obv_str, f_col(obv_str)) f_cell(5, "ATR", atr_str, col_neu)

使用說明

  • 貼到 TradingView → Pine 編輯器 → 儲存 → 加到圖表
  • EMA、VWAP、RSI 同時顯示多▲ = 趨勢確認，可加大部位
  • ADX < 25 顯示「震盪」，此時趨勢訊號準確率降低

//@version=5 indicator("即時動能訊號欄", overlay=true) rsi_len = input.int(14, "RSI 週期") vol_ma = input.int(20, "成交量均線") rsi = ta.rsi(close, rsi_len) ema9 = ta.ema(close, 9) ema21 = ta.ema(close, 21) vol_avg = ta.sma(volume, vol_ma) obv = ta.obv atr = ta.atr(14) // 動能方向 momentum = close - close[3] mom_str = momentum > 0 ? "上▲" : "下▼" // RSI 評估（顯示數值+評估） rsi_val = math.floor(rsi * 100) / 100 rsi_disp = str.tostring(rsi_val) + (rsi >= 70 ? "(超買)" : rsi <= 30 ? "(超賣)" : rsi >= 50 ? "(偏多)" : "(偏空)") // EMA 多空 ema_str = ema9 > ema21 ? "多▲" : "空▼" // 成交量 vol_str = volume > vol_avg * 1.5 ? "爆量!" : volume > vol_avg ? "放量" : "縮量" // OBV 方向 obv_str = obv > obv[1] ? "買▲" : "賣▼" // ATR atr_disp = str.tostring(math.floor(atr * 100) / 100) col_g = color.new(color.green, 0) col_r = color.new(color.red, 0) col_y = color.new(color.yellow,0) col_w = color.new(color.white, 0) col_gr= color.new(color.gray, 0) var t = table.new(position.bottom_right, 2, 6, bgcolor=#161b22, border_color=#30363d, border_width=1, frame_color=#30363d, frame_width=1) f_cell(r, lbl, val, vc) => table.cell(t, 0, r, lbl, text_color=#8b949e, text_size=size.small, bgcolor=#1a1f2e) table.cell(t, 1, r, val, text_color=vc, text_size=size.small, bgcolor=#161b22) f_cell(0, "動能", mom_str, momentum > 0 ? col_g : col_r) f_cell(1, "RSI", rsi_disp, rsi >= 70 ? col_r : rsi <= 30 ? col_g : rsi >= 50 ? col_g : col_r) f_cell(2, "EMA", ema_str, ema9 > ema21 ? col_g : col_r) f_cell(3, "成交量", vol_str, vol_str == "爆量!" ? col_y : vol_str == "放量" ? col_g : col_gr) f_cell(4, "OBV", obv_str, obv > obv[1] ? col_g : col_r) f_cell(5, "ATR", atr_disp, col_w)

使用說明

  • 動能＋EMA＋OBV 三者同向 = 順勢進場強訊號
  • 「爆量!」出現在關鍵價位附近 = 機構介入訊號
  • RSI 超買 + 動能下▼ = 短線轉弱，考慮減碼
跳空區間
跳空壓力 / 支撐區間框
自動偵測 K 棒跳空缺口，用紅框（跳空壓力）與綠框（跳空支撐）標示在圖上，缺口填補後自動消失
適用標的
所有標的
偵測邏輯
K棒缺口（非日缺口）
版本
Pine Script v5
顯示
圖表上方/下方方框
//@version=5 indicator("跳空壓力支撐區間", overlay=true, max_boxes_count=200) min_gap_pct = input.float(0.1, "最小缺口比例(%)", minval=0.01, step=0.05) / 100 max_bars = input.int(100, "最多保留幾根K棒", minval=10) col_up_gap = input.color(color.new(color.green, 80), "跳空支撐填色") col_dn_gap = input.color(color.new(color.red, 80), "跳空壓力填色") border_up = input.color(color.green, "跳空支撐邊框") border_dn = input.color(color.red, "跳空壓力邊框") // 跳空向上（支撐缺口）: 本K低點 > 前K高點 gap_up = low > high[1] and (low - high[1]) / high[1] >= min_gap_pct // 跳空向下（壓力缺口）: 本K高點 < 前K低點 gap_dn = high < low[1] and (low[1] - high) / low[1] >= min_gap_pct if gap_up box.new(bar_index - 1, low, bar_index + max_bars, high[1], bgcolor=col_up_gap, border_color=border_up, border_width=1) if gap_dn box.new(bar_index - 1, low[1], bar_index + max_bars, high, bgcolor=col_dn_gap, border_color=border_dn, border_width=1) // 缺口填補後延伸框至當前 bar var boxes = array.new_box() array.push(boxes, gap_up ? box.new(bar_index,low,bar_index,high[1],border_color=color.na) : box.na) plotshape(gap_up, style=shape.triangleup, location=location.belowbar, color=color.green, size=size.tiny, title="跳空支撐") plotshape(gap_dn, style=shape.triangledown, location=location.abovebar, color=color.red, size=size.tiny, title="跳空壓力")

使用說明

  • 綠框 = 跳空支撐（回測此區域時考慮買進）
  • 紅框 = 跳空壓力（上漲遇此區域考慮減碼）
  • 調整「最小缺口比例」可過濾雜訊（建議 0.1–0.3%）
  • 「最多保留幾根K棒」決定缺口框延伸長度

//@version=5 indicator("多週期多空判斷（DMI）", overlay=true) adx_len = input.int(14, "ADX 週期") adx_thr = input.int(25, "趨勢強度門檻") f_dmi(tf) => [di_p, di_m, adx_v] = request.security(syminfo.tickerid, tf, ta.dmi(adx_len, adx_len), lookahead=barmerge.lookahead_off) label = di_p > di_m ? "多▲" : "空▼" strength = adx_v >= adx_thr ? (di_p > di_m ? "強多" : "強空") : "震盪" adx_disp = str.tostring(math.floor(adx_v * 100) / 100) full = strength + " ADX:" + adx_disp col = di_p > di_m ? color.new(color.green,0) : color.new(color.red,0) [full, col] [s5m, c5m] = f_dmi("5") [s30m, c30m] = f_dmi("30") [s1d, c1d] = f_dmi("D") [s5d, c5d] = f_dmi("5D") [s10d, c10d] = f_dmi("10D") [s20d, c20d] = f_dmi("20D") var t = table.new(position.top_left, 2, 7, bgcolor=#161b22, border_color=#30363d, border_width=1, frame_color=#30363d, frame_width=1) f_cell(r, lbl, val, vc) => table.cell(t, 0, r, lbl, text_color=#8b949e, text_size=size.small, bgcolor=#1a1f2e) table.cell(t, 1, r, val, text_color=vc, text_size=size.small, bgcolor=#161b22) table.cell(t, 0, 0, "週期", text_color=#e6edf3, bgcolor=#21262d, text_size=size.small) table.cell(t, 1, 0, "多空/ADX", text_color=#e6edf3, bgcolor=#21262d, text_size=size.small) f_cell(1, "5分", s5m, c5m) f_cell(2, "30分", s30m, c30m) f_cell(3, "日", s1d, c1d) f_cell(4, "5日", s5d, c5d) f_cell(5, "10日", s10d, c10d) f_cell(6, "20日", s20d, c20d)

使用說明

  • 多個週期同時顯示「強多」= 趨勢共振，勝率最高
  • ADX < 25 顯示「震盪」，此時不追趨勢單
  • 短中長週期「強多→震盪→空▼」= 趨勢正在弱化，準備出場
